/**
 * @fileoverview Controller de Autenticação da aplicação MVC631.
 * Gerencia o fluxo de login, logout e controle de sessão dos usuários,
 * interceptando as requisições HTTP das rotas de autenticação.
 * @module authController
 */

/**
 * @typedef {Object} Request
 * @description Objeto de Requisição do Express. Contém dados enviados pelo cliente
 * como `req.body`, `req.params`, `req.query` e `req.session`.
 */

/**
 * @typedef {Object} Response
 * @description Objeto de Resposta do Express. Utilizado para enviar respostas HTTP
 * ao cliente através de métodos como `res.render()`, `res.redirect()` e `res.status()`.
 */

const User = require("../models/User");

/**
 * Renderiza a página de login.
 * Redireciona para a home caso o usuário já possua uma sessão ativa,
 * evitando acesso redundante à tela de login.
 * Os cabeçalhos de cache são configurados para impedir que o navegador
 * armazene a página e permita retornar a ela com o botão "Voltar".
 * @param {Request} req - Objeto de Requisição do Express.
 *   Verifica `req.session.user` para detectar sessão ativa.
 * @param {Response} res - Objeto de Resposta do Express.
 * @returns {void} Renderiza a view `login` ou redireciona para `/` se já autenticado.
 */
exports.getLogin = (req, res) => {
    res.set("Cache-Control", "no-store, no-cache, must-revalidate, private");
    res.set("Pragma", "no-cache");
    res.set("Expires", "0");

    if (req.session.user) {
        return res.redirect("/");
    }

    res.render("login", {
        title: "Login",
        error: null
    });
};

/**
 * Processa as credenciais enviadas pelo formulário de login.
 * Valida o e-mail e a senha contra o Model de usuários. Em caso de sucesso,
 * persiste os dados do usuário na sessão e redireciona para a home.
 * Em caso de falha, retorna HTTP 401 e renderiza novamente a view com mensagem de erro.
 * @param {Request} req - Objeto de Requisição do Express.
 *   Espera os campos `email` e `senha` no corpo da requisição (`req.body`).
 * @param {Response} res - Objeto de Resposta do Express.
 * @returns {void} Redireciona para `/` em caso de sucesso, ou renderiza `login`
 *   com status HTTP 401 e mensagem de erro em caso de falha.
 * @throws {Error} Não dispara erros diretos, mas depende de `User.findByEmail` para validação.
 * @todo Implementar hash de senha com bcrypt e substituir comparação direta de strings.
 */
exports.login = (req, res) => {
    const { email, senha } = req.body;
    const user = User.findByEmail(email);

    if (!user || user.senha !== senha) {
        return res.status(401).render("login", {
            title: "Login",
            error: "E-mail ou senha invalido."
        });
    }

    req.session.user = {
        id: user.id,
        nome: user.nome,
        email: user.email
    };

    res.redirect("/");
};

/**
 * Encerra a sessão do usuário autenticado e redireciona para a tela de login.
 * Destrói a sessão do lado do servidor e limpa o cookie `connect.sid` no cliente.
 * Os cabeçalhos de cache são configurados para evitar que páginas autenticadas
 * fiquem acessíveis via histórico do navegador após o logout.
 * @param {Request} req - Objeto de Requisição do Express.
 *   Utiliza `req.session.destroy()` para invalidar a sessão ativa.
 * @param {Response} res - Objeto de Resposta do Express.
 * @returns {void} Redireciona para `/login` após o logout bem-sucedido,
 *   ou para `/` em caso de erro na destruição da sessão.
 */
exports.logout = (req, res) => {
    req.session.destroy(error => {
        if (error) {
            return res.redirect("/");
        }

        res.clearCookie("connect.sid");
        res.set("Cache-Control", "no-store, no-cache, must-revalidate, private");
        res.set("Pragma", "no-cache");
        res.set("Expires", "0");
        res.redirect("/login");
    });
};
