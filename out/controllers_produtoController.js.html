/**
 * @fileoverview Controller de Produtos da aplicação MVC631.
 * Intercepta as requisições HTTP relacionadas à entidade Produto e
 * orquestra as respostas utilizando as funções do Model correspondente.
 * @module produtoController
 */

/**
 * @typedef {Object} Request
 * @description Objeto de Requisição do Express. Contém dados enviados pelo cliente
 * como `req.body`, `req.params`, `req.query` e `req.session`.
 */

/**
 * @typedef {Object} Response
 * @description Objeto de Resposta do Express. Utilizado para enviar respostas HTTP
 * ao cliente através de métodos como `res.render()`, `res.redirect()` e `res.status()`.
 */

const {
    Produto,
    produtos,
    gerarProximoProdutoId,
    converterPrecoParaCentavos,
    formatarPreco,
    formatarPrecoParaInput
} = require("../models/produtoModel");

/**
 * Lista todos os produtos cadastrados e renderiza a view de produtos.
 * @param {Request} req - Objeto de Requisição do Express. Não exige parâmetros adicionais.
 * @param {Response} res - Objeto de Resposta do Express.
 * @returns {void} Renderiza a view `produtos` com a lista completa e a função de formatação de preço.
 */
exports.getProdutos = (req, res) => {
    res.render("produtos", { produtos, formatarPreco });
};

/**
 * Recebe os dados do formulário, cria um novo produto e redireciona para a listagem.
 * O preço recebido via `req.body` é automaticamente convertido para centavos antes do armazenamento.
 * @param {Request} req - Objeto de Requisição do Express.
 *   Espera os campos `nome`, `descricao` e `preco` no corpo da requisição (`req.body`).
 * @param {Response} res - Objeto de Resposta do Express.
 * @returns {void} Redireciona para `/produtos` após a criação bem-sucedida.
 */
exports.createProduto = (req, res) => {
    const { nome, descricao, preco } = req.body;
    const precoEmCentavos = converterPrecoParaCentavos(preco);
    const novoProduto = new Produto(gerarProximoProdutoId(), nome, descricao, precoEmCentavos);
    produtos.push(novoProduto);
    res.redirect("/produtos");
};

/**
 * Busca um produto pelo ID e renderiza a view de edição com os dados preenchidos.
 * Retorna status 404 caso o produto não seja encontrado na lista em memória.
 * @param {Request} req - Objeto de Requisição do Express.
 *   Espera o parâmetro `id` na URL (`req.params.id`).
 * @param {Response} res - Objeto de Resposta do Express.
 * @returns {void} Renderiza a view `editProduto` com o produto e a função de formatação para input,
 *   ou retorna HTTP 404 caso o produto não exista.
 */
exports.getEditProduto = (req, res) => {
    const { id } = req.params;
    const produto = produtos.find(p => p.id == id);
    if (!produto) {
        return res.status(404).send("Produto não encontrado");
    }
    res.render("editProduto", { produto, formatarPrecoParaInput });
};

/**
 * Atualiza os dados de um produto existente com base no ID recebido via URL.
 * Se o produto não for localizado, nenhuma alteração é realizada (fail silently).
 * @param {Request} req - Objeto de Requisição do Express.
 *   Espera o parâmetro `id` em `req.params` e os campos `nome`, `descricao` e `preco` em `req.body`.
 * @param {Response} res - Objeto de Resposta do Express.
 * @returns {void} Redireciona para `/produtos` após o processamento.
 * @todo Adicionar retorno de erro HTTP 404 explícito quando o produto não for encontrado.
 */
exports.updateProduto = (req, res) => {
    const { id } = req.params;
    const { nome, descricao, preco } = req.body;
    const produto = produtos.find(p => p.id == id);

    if (produto) {
        produto.nome = nome;
        produto.descricao = descricao;
        produto.preco = converterPrecoParaCentavos(preco);
    }

    res.redirect("/produtos");
};

/**
 * Remove um produto da lista em memória com base no seu ID.
 * A remoção é feita por `splice` no índice encontrado via `findIndex`.
 * Se o produto não existir, nenhuma ação é tomada.
 * @param {Request} req - Objeto de Requisição do Express.
 *   Espera o parâmetro `id` na URL (`req.params.id`).
 * @param {Response} res - Objeto de Resposta do Express.
 * @returns {void} Redireciona para `/produtos` após a remoção.
 */
exports.deleteProduto = (req, res) => {
    const { id } = req.params;
    const index = produtos.findIndex(p => p.id == id);

    if (index !== -1) {
        produtos.splice(index, 1);
    }

    res.redirect("/produtos");
};

/**
 * Renderiza a página estática "Sobre Nós" da aplicação.

exports.getAbout = (req, res) => {
    res.render("about", { title: "Sobre Nós" });
