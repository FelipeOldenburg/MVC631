/**
 * @class Produto
 * @classdesc Representa a entidade Produto e define a estrutura de dados utilizada
 * em memória para persistência temporária no padrão MVC da aplicação.
 */
class Produto {
    /**
     * Cria uma nova instância de Produto.
     * @param {number} id - Identificador único do produto (gerado automaticamente).
     * @param {string} nome - Nome comercial do produto.
     * @param {string} descricao - Descrição detalhada do produto.
     * @param {number} preco - Preço do produto armazenado em centavos (ex: R$ 350,00 → 35000).
     */
    constructor(id, nome, descricao, preco) {
        this.id = id;
        this.nome = nome;
        this.descricao = descricao;
        this.preco = preco;
    }
}

/**
 * Lista em memória contendo todos os produtos cadastrados na aplicação.
 * @type {Produto[]}
 */
let produtos = [
    new Produto(1, "Notebook", "Notebook para desenvolvimento", 350000),
    new Produto(2, "Mouse", "Mouse sem fio", 8500),
    new Produto(3, "Teclado", "Teclado mecanico RGB", 45000)
];

/**
 * Contador interno que rastreia o próximo ID disponível para novos produtos.
 * Inicializado com base no maior ID existente na lista + 1.
 * @type {number}
 */
let proximoProdutoId = Math.max(...produtos.map(produto => produto.id), 0) + 1;

/**
 * Gera e retorna o próximo ID único disponível para um novo produto.
 * Incrementa o contador interno a cada chamada.
 * @returns {number} O próximo ID sequencial disponível.
 */
function gerarProximoProdutoId() {
    return proximoProdutoId++;
}

/**
 * Converte um valor de preço (string formatada em pt-BR ou número float) para centavos inteiros.
 * Garante consistência no armazenamento, evitando erros de ponto flutuante.
 * @param {string|number} preco - Preço no formato brasileiro (ex: "3.500,00" ou 35.00) ou número float.
 * @returns {number} Valor inteiro equivalente em centavos (ex: 3500000 ou 3500).
 *
 * @example
 * converterPrecoParaCentavos("3.500,00"); // Retorna 350000
 * converterPrecoParaCentavos(35.00);      // Retorna 3500
 */
function converterPrecoParaCentavos(preco) {
    if (typeof preco === "number") {
        return Math.round(preco * 100);
    }

    const precoLimpo = String(preco).replace(/[^\d,.]/g, "");
    const precoNormalizado = precoLimpo.includes(",")
        ? precoLimpo.replace(/\./g, "").replace(",", ".")
        : precoLimpo;

    return Math.round(Number(precoNormalizado) * 100);
}

/**
 * Formata um valor em centavos para exibição no padrão de moeda brasileira (BRL).
 * @param {number} precoEmCentavos - Valor inteiro em centavos (ex: 350000).
 * @returns {string} Preço formatado como string no padrão pt-BR (ex: "R$ 3.500,00").
 *
 * @example
 * formatarPreco(350000); // Retorna "R$ 3.500,00"
 */
function formatarPreco(precoEmCentavos) {
    return (precoEmCentavos / 100).toLocaleString("pt-BR", {
        style: "currency",
        currency: "BRL"
    });
}

/**
 * Formata um valor em centavos para o padrão aceito por inputs do tipo texto (pt-BR sem símbolo).
 * Utilizado para pré-preencher formulários de edição com o valor atual do produto.
 * @param {number} precoEmCentavos - Valor inteiro em centavos (ex: 350000).
 * @returns {string} Preço formatado para input (ex: "3500,00").
 *
 * @example
 * formatarPrecoParaInput(350000); // Retorna "3500,00"
 */
function formatarPrecoParaInput(precoEmCentavos) {
    return (precoEmCentavos / 100).toFixed(2).replace(".", ",");
}
